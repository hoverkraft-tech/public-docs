export const projectSectionsWithIds = (() => {
  const createSlug = (value) =>
    value
      .toLowerCase()
      .trim()
      .replace(/&/g, "and")
      .replace(/[^a-z0-9\s-]/g, "")
      .replace(/\s+/g, "-")
      .replace(/-+/g, "-");

  const seen = new Map();

  return projectSections.map((section) => {
    const baseId = createSlug(section.title);
    const nextIndex = (seen.get(baseId) ?? 0) + 1;
    seen.set(baseId, nextIndex);

    const id = nextIndex === 1 ? baseId : `${baseId}-${nextIndex}`;
    return { ...section, id };
  });
})();
